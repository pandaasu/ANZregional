Oracle 9i Java Security - Based on Java 2 Security Model
--------------------------------------------------------

java.io.FilePermission
----------------------
<<ALL FILES>> = any files
'/path/*' = all files in the specified path
'/path/-' = all file and recursively all sub directories and files in the specified path


execute dbms_java.grant_permission('LICS_APP','java.io.FilePermission','/ics/-','read,write,execute,delete');
execute dbms_java.grant_permission('LICS_APP','java.lang.RuntimePermission','*','readFileDescriptor,writeFileDescriptor');
commit;

create directory ics_binary as '/ics/bin';
create directory ics_inbound_data as '/ics/inbound/data';
create directory ics_inbound_archive as '/ics/inbound/archive';
create directory ics_outbound_data as '/ics/outbound/data';
create directory ics_outbound_archive as '/ics/outbound/archive';
create directory ics_passthru as '/ics/passthru';

grant all on directory ics_binary to lics_app;
grant all on directory ics_inbound_data to lics_app;
grant all on directory ics_inbound_archive to lics_app;
grant all on directory ics_outbound_data to lics_app;
grant all on directory ics_outbound_archive to lics_app;
grant all on directory ics_passthru to lics_app;
